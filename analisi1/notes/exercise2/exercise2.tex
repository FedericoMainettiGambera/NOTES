\[
    x^2 +4x +4 +y^2 < x^2+y^2

\[
    x>1
\]

\[

\]
\[
    5-\frac{2}{n} < 3 +\epsilon

\[
    \frac{2}{n} > 2 -\epsilon
\]
\[
    n < \frac{2}{2-\epsilon}
\]
